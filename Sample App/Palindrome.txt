#include <stdio.h>
#include <stdbool.h>
#include <string.h>
#include <ctype.h>

bool IsPalindrome(char f_str[])
{
    int l_count = 0;
    int len = strlen(f_str) - 1;
    if(len<2)
    {
       return false;
    }
    while(len > 1)
    {
        if(tolower(f_str[l_count++]) != tolower(f_str[len--]))
        {
            return false;
        }
    }
    return true;
}

int main()
{
    // Change l_str to check for palindrome
    char l_str[] = "Dad";
    if(IsPalindrome(l_str))
    {
        printf("%s- String is a palindrome\n", l_str);
    }
    else
    {
        printf("%s- String is not a palindrome\n", l_str);
    }
    
    return 0;
}
