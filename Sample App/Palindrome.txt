#include <stdio.h>
#include <stdbool.h>
#include <string.h>

bool IsPalindrome(char f_str[])
{
    int l_count = 0;
    int len = strlen(f_str) - 1;
    if(len<2)
    {
       return false;
    }
    while(len > 1)
    {
        if(tolower(f_str[l_count++]) != tolower(f_str[len--]))
        {
            return false;
        }
    }
    return true;
}

int main()
{
    if(IsPalindrome("1881"))
    {
      printf("1881 - String is a palindrome\n");
    }
    else
    {
        printf("1881 - String is not a palindrome\n");
    }
    
    if(IsPalindrome("Dad"))
    {
        printf("Dad - String is a palindrome\n");
    }
    else
    {
        printf("Dad - String is not a palindrome\n");
    }
    
    if(IsPalindrome("Step on no pets"))
    {
        printf("Step on no pets - String is a palindrome\n");
    }
    else
    {
        printf("Step on no pets - String is not a palindrome\n");
    }
    
    if(IsPalindrome("Hello World"))
    {
        printf("Hello World - String is a palindrome\n");
    }
    else
    {
        printf("Hello World - String is not a palindrome\n");
    }
    return 0;
}
