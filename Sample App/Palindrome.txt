#include <stdio.h>
#include <stdbool.h>
#include <string.h>

bool IsPalindrome(char f_str[])
{
    int l_count = 0;
    int len = strlen(f_str) - 1;
    if(len<2)
    {
       return false;
    }
    while(len > 1)
    {
        if(tolower(f_str[l_count++]) != tolower(f_str[len--]))
        {
            return false;
        }
    }
    return true;
}

int main()
{
    // Change string in IsPalindrome() to check for palindrome
    if(IsPalindrome("Dad"))
    {
      printf("String is a palindrome\n");
    }
    else
    {
        printf("String is not a palindrome\n");
    }
    
    return 0;
}
